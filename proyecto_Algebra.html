📐 Código del Modelo Dinámico

import numpy as np
from scipy.integrate import odeint

# Parámetros del modelo (obtenidos del análisis)
lambda_VP = 0.6057  # Visita → Producto
lambda_PC = 0.1638  # Producto → Carrito
lambda_CB = 0.0163  # Carrito → Compra
lambda_CA = 0.10    # Carrito → Abandono

# Sistema de ecuaciones diferenciales
def funnel_system(y, t, lambda_VP, lambda_PC, lambda_CB, lambda_CA):
    """
    Modelo dinámico del funnel de conversión ecommerce
    
    Variables de estado:
    - V(t): Visitas
    - P(t): Producto visto
    - C(t): Carrito
    - B(t): Compras realizadas
    - A(t): Abandonos
    """
    V, P, C, B, A = y
    
    # Ecuaciones diferenciales
    dVdt = -lambda_VP * V
    dPdt = lambda_VP * V - lambda_PC * P
    dCdt = lambda_PC * P - (lambda_CB + lambda_CA) * C
    dBdt = lambda_CB * C
    dAdt = lambda_CA * C
    
    return [dVdt, dPdt, dCdt, dBdt, dAdt]

# Condiciones iniciales
V0 = 10000   # visitas iniciales
P0 = 0       # sin usuarios en producto
C0 = 0       # sin usuarios en carrito
B0 = 0       # sin compras
A0 = 0       # sin abandonos

y0 = [V0, P0, C0, B0, A0]

# Tiempo de simulación
t = np.linspace(0, 50, 1000)

# Resolver sistema
solution = odeint(
    funnel_system,
    y0,
    t,
    args=(lambda_VP, lambda_PC, lambda_CB, lambda_CA)
)

V, P, C, B, A = solution.T
